<script type="text/javascript">
    $(document).ready(function () {        
        var oWPSlider = {
            images: $("#WPSlider img"),
            iPos: 0,
            width: 103,
            iCurrentMoved: 0
        };
        var oW8Slider = {
            images: $("#Win8Slider img"),
            iPos: 0,
            width: 103,
            iCurrentMoved: 0
        };
        var oOfficeSlider = {
            images: $("#OfficeContent img"),
            iPos: 0,
            width: 103,
            iCurrentMoved: 0
        };
        oWPSlider.iTotalImages = oWPSlider.images.length;
        oW8Slider.iTotalImages = oW8Slider.images.length;
        oOfficeSlider.iTotalImages = oOfficeSlider.images.length;
        function SetImagePosition(oSlider) {
            oSlider.images.each(function (index) {
                $(this).css("left", oSlider.iPos + "px").attr("data-index", index);
                oSlider.iPos += oSlider.width;
            });
        }
        function MoveLeft(oSlider) {
            if (!oSlider.images.is(":animated")) {
                oSlider.iCurrentMoved--;
                if (oSlider.iCurrentMoved < 0) {
                    oSlider.iCurrentMoved = oWPSlider.iTotalImages + oSlider.iCurrentMoved;
                }
                $(oSlider.images[oSlider.iCurrentMoved]).css("left", "-=" + (oWPSlider.iTotalImages * oSlider.width) + "px");
                oSlider.images.animate({ left: "+=" + oSlider.width + "px" }, 300);
            }
        }
        function MoveRight(oSlider) {
            if (!oSlider.images.is(":animated")) {
                oSlider.images.animate({ left: "-=" + oSlider.width + "px" }, 300);
                $(oSlider.images).promise().done(function () {
                    if (oSlider.iCurrentMoved >= oSlider.iTotalImages) {
                        oSlider.iCurrentMoved = 0;
                    }
                    $(oSlider.images[oSlider.iCurrentMoved]).css("left", "+=" + (oSlider.iTotalImages * oSlider.width) + "px");
                    oSlider.iCurrentMoved++;
                });
            }
        }
        $("#WinPhoneLeftArrow").click(function () {
            MoveLeft(oWPSlider);
        });
        $("#WinPhoneRightArrow").click(function () {
            MoveRight(oWPSlider);
        });
        $("#Win8LeftArrow").click(function () {
            MoveLeft(oW8Slider);
        });
        $("#Win8RightArrow").click(function () {
            MoveRight(oW8Slider);
        });
        $("#OfficeLeftArrow").click(function () {
            MoveLeft(oOfficeSlider);
        });
        $("#OfficeRightArrow").click(function () {
            MoveRight(oOfficeSlider);
        });
        function InitAppSlider() {
            SetImagePosition(oWPSlider);
            SetImagePosition(oW8Slider);
            SetImagePosition(oOfficeSlider);
        }                    
        InitAppSlider();
    });
</script>
